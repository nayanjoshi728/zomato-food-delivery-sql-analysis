   Records: 30
   
3. **orders** table

            total_amount, order_date, order_time
   Records: 7,100
   
4. **deliveries** table
   Columns: delivery_id (PK), order_id (FK), rider_id (FK), delivery_time, 
            delivery_status, delivery_date
   Records: 5,693
   
5. **riders** table
   Columns: rider_id (PK), rider_name, phone, rating, city, vehicle_type
   Records: 50

# DATA DEFINITION LANGUAGE (DDL) REQUIREMENTS
# ==================================================================================

## Indexes (Recommended for Performance)
CREATE INDEX idx_order_customer ON orders(customer_id);
CREATE INDEX idx_order_restaurant ON orders(restaurant_id);
CREATE INDEX idx_order_date ON orders(order_date);
CREATE INDEX idx_delivery_rider ON deliveries(rider_id);
CREATE INDEX idx_delivery_status ON deliveries(delivery_status);

## Foreign Key Constraints (Data Integrity)
ALTER TABLE orders ADD CONSTRAINT fk_order_customer 
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id);
    
ALTER TABLE orders ADD CONSTRAINT fk_order_restaurant 
    FOREIGN KEY (restaurant_id) REFERENCES restaurants(restaurant_id);
    
ALTER TABLE deliveries ADD CONSTRAINT fk_delivery_order 
    FOREIGN KEY (order_id) REFERENCES orders(order_id);
    
ALTER TABLE deliveries ADD CONSTRAINT fk_delivery_rider 
    FOREIGN KEY (rider_id) REFERENCES riders(rider_id);

# ==================================================================================
# TROUBLESHOOTING
# ==================================================================================

## Issue: "Access Denied for user 'root'@'localhost'"
Solution: Verify MySQL is running, check password, reset if needed

## Issue: "Database does not exist"
Solution: Run CREATE DATABASE zomato_project; first

## Issue: "Table already exists"
Solution: Run DROP TABLE IF EXISTS [table_name]; first

## Issue: "Foreign key constraint fails"
Solution: Ensure parent tables are created before child tables

## Issue: Query returns empty results
Solution: Verify sample data was imported correctly

## Issue: TIMESTAMPDIFF returns NULL or 0
Solution: Check delivery_time and order_time formats are DATETIME


# ==================================================================================
# PERFORMANCE OPTIMIZATION TIPS
# ==================================================================================

1. Create indexes on frequently queried columns (customer_id, restaurant_id, order_date)
2. Use EXPLAIN to analyze query performance
3. Partition large tables by date if dataset grows beyond 1GB
4. Use LIMIT clause during development to reduce result set size
5. Archive old data (pre-2024) to separate tables if needed
